async function submitData() { const input = document.getElementById('userInput'); const status = document.getElementById('status'); if (!input.value) return; status.style.display = 'block'; status.className = 'loading'; status.innerText = 'กำลังประมวลผลด้วย AI...'; try { const response = await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify({ source: 'web', message: input.value }) }); const result = await response.json(); if (result.status === "success" && result.data) { status.className = 'success'; status.innerText = '✅ บันทึกสำเร็จ!\n' + result.data.map(i => `- ${i.item}: ${i.amount}บ.`).join('\n'); input.value = ''; } else { // หาก AI ส่ง Error กลับมา ให้แสดงข้อความเตือนที่ได้รับ status.style.backgroundColor = '#f8d7da'; status.style.color = '#721c24'; status.innerText = '❌ ' + (result.message | | 'AI ไม่สามารถประมวลผลได้'); } } catch (e) { status.innerText = '❌ เกิดข้อผิดพลาดในการเชื่อมต่อ: ' + e; } }
